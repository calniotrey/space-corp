	id = space_corps_diplomacy.10
	is_triggered_only = yes

	}

country_event = {
	id = space_corps_diplomacy.20
	title = space_corps_diplomacy.20.name
	desc = space_corps_diplomacy.20.desc
	show_sound = event_conversation
	diplomatic = yes
	is_triggered_only = yes
	picture_event_data = {
		portrait = event_target:space_corps_country
		room = ethic_spaceship_room
	}
	trigger = {
		exists = event_target:space_corps_country
	}
	option = {
		name = space_corps_diplomacy.20.hire_scientist
		hidden_effect = {
			if = {
				limit = {
					has_country_resource = {
						type = energy
						amount > 4095
					}
				}
				SLEX_get_custom_parameter_advanced = {
					CONDITION_NAME = has_country_resource
					FIRST_KW = type
					SECOND_KW = energy
					THIRD_KW = amount
					VARIABLE = space_corps_energy_stock
					ROUNDING_ZEROES = "00"
				}
			}
			else = {
				SLEX_get_custom_parameter_advanced = {
					CONDITION_NAME = has_country_resource
					FIRST_KW = type
					SECOND_KW = energy
					THIRD_KW = amount
					VARIABLE = space_corps_energy_stock
					ROUNDING_ZEROES = "0"
				}
			}
			event_target:space_corps_country = {
				select_5_random_recruitable_leaders = {
					CLASS = scientist
				}
			}
		}
		country_event = {
			id = space_corps_diplomacy.21
		}
	}
	option = {
		# back
		name = space_corps_diplomacy.back
		default_hide_option = yes
		hidden_effect = {
			country_event = {
				id = space_corps_diplomacy.11
			}
		}
	}
	option = {
		# Quit
		name = space_corps_diplomacy.quit
	}
}
