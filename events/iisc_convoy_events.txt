namespace = iisc_convoy
# reset convoy orders
country_event = {
	id = iisc_convoy.10
	is_triggered_only = yes
	hide_window = yes
	immediate = {
		if = {
			limit = {
				exists = event_target:iisc_convoy_fleet
			}
			event_target:iisc_convoy_fleet = {
				iisc_set_convoy_orders = yes
			}
		}
	}
}

# set name
fleet_event = {
	id = iisc_convoy.20
	is_triggered_only = yes
	hide_window = yes
	immediate = {
		set_name = "[iisc_hq_planet.GetName] -> [iisc_convoy_dest.GetName] [This.GetName]"
	}
}
