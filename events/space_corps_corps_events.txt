# This = Corp
# on_tech_increased
country_event = {
	id = space_corps_corps.31
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		is_country_type = space_corp
	}
	immediate = {
		if = {
			limit = {
				exists = overlord
			}
			overlord = {
				copy_techs_from = {
					target = prev
					except = {
						# not dangerous techs
						tech_synthetic_workers
						tech_combat_computers_3
						tech_jump_drive_1
						tech_psi_jump_drive_1
					}
				}
			}
		}
	}
}
