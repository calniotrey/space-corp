namespace = space_corps_start
country_event = {
	id = space_corps_start.10
	is_triggered_only = yes
	hide_window = yes
	trigger = {
		is_country_type = default
		NOT = {
			has_country_flag = space_corps_initialized
		}
	}
	immediate = {
		set_country_flag = space_corps_initialized
		set_policy = {
			policy = space_corps_taxes_policy
			option = space_corps_no_taxes
			cooldown = no
		}
		set_policy = {
			policy = space_corps_license_cost_policy
			option = space_corps_no_license_cost
			cooldown = no
		}
		set_policy = {
			policy = space_corps_survey_reward_policy
			option = space_corps_survey_no_reward
			cooldown = no
		}
		# to make sure that the policy is changed => trigger the effect field
		set_policy = {
			policy = space_corps_taxes_policy
			option = space_corps_medium_taxes
			cooldown = no
		}
		set_policy = {
			policy = space_corps_license_cost_policy
			option = space_corps_low_license_cost
			cooldown = no
		}
		set_policy = {
			policy = space_corps_survey_reward_policy
			option = space_corps_survey_medium_reward
			cooldown = no
		}
		randomize_space_corps_spawn_delay = yes
		# debug
		set_variable = {
			which = years_until_next_corp_creation
			value = 0
		}
	}
}
