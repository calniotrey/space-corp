defined_text = {
	name = GetHeadquartersPlanet
	text = {
		trigger = {
			exists = event_target:headquarters_planet
		}
		localization_key = "[headquarters_planet.GetName]"
	}
	text = {
		trigger = {
			NOT = {
				exists = event_target:headquarters_planet
			}
		}
		localization_key = "Unknown planet"
	}
}
