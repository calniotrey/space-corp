iisc_corp_spawning_all_on = {
	potential = {
		NOT = { has_global_flag = iisc_corp_spawning_all }
	}
	allow = {
		always = yes
	}
	
	effect = {
		set_global_flag = iisc_corp_spawning_all
		remove_global_flag = iisc_corp_spawning_players
		remove_global_flag = iisc_corp_spawning_ai
	}
}

	potential = {
		has_global_flag = iisc_corp_spawning_all
	}

	}
}
